170  subroutine solve_circulant(self, eigvals, rhs, res)
171  class(sll_t_maxwell_1d_fem) :: self
172  sll_real64, intent(in) :: eigvals(:) ! eigenvalues of circulant matrix
173  sll_real64, intent(in) :: rhs(:)
174  sll_real64, intent(out) :: res(:)
175  ! local variables
176  sll_int32 :: k
177  sll_real64 :: re, im
178 
179  ! Compute res from rhs, using eigenvalue of circulant matrix
180  res = rhs
181  ! Forward FFT
182  call sll_s_fft_exec_r2r_1d(self%plan_fw, res, self%wsave)
183  self%wsave(1) = self%wsave(1)*eigvals(1)
184  do k = 2, self%n_dofs/2
185  re = self%wsave(k)*eigvals(k) - &
186  self%wsave(self%n_dofs - k + 2)*eigvals(self%n_dofs - k + 2)
187  im = self%wsave(k)*eigvals(self%n_dofs - k + 2) + &
188  self%wsave(self%n_dofs - k + 2)*eigvals(k)
189  self%wsave(k) = re
190  self%wsave(self%n_dofs - k + 2) = im
191  end do
192  self%wsave(self%n_dofs/2 + 1) = self%wsave(self%n_dofs/2 + 1)*eigvals(self%n_dofs/2 + 1)
193  ! Backward FFT
194  call sll_s_fft_exec_r2r_1d(self%plan_bw, self%wsave, res)
195  ! normalize

322  subroutine init_1d_fem(self, domain, n_dofs, s_deg_0)
323  class(sll_t_maxwell_1d_fem), intent(out) :: self
324  sll_real64, intent(in) :: domain(2) ! xmin, xmax
325  sll_int32, intent(in) :: n_dofs ! number of degrees of freedom (here number of cells and grid points)

327  sll_int32, intent(in) :: s_deg_0 ! highest spline degree
328 
329  ! local variables
330  sll_int32 :: ierr

332  sll_real64 :: coef0, coef1, sin_mode, cos_mode
333 
334  self%n_dofs = n_dofs

341  sll_allocate(self%mass_1(s_deg_0), ierr)
342 
343  select case (s_deg_0)
344  case (1) ! linear and constant splines
345  ! Upper diagonal coeficients of linear spline mass matrix (from Eulerian numbers)
346  self%mass_0(1) = 4.0_f64/6.0_f64
347  self%mass_0(2) = 1.0_f64/6.0_f64
348  ! Upper diagonal coeficients of constant spline mass matrix
349  self%mass_1(1) = 1.0_f64
350  case (2) ! quadratic and linear splines
351  ! Upper diagonal coeficients of quadratic spline mass matrix (from Eulerian numbers)
352  self%mass_0(1) = 66.0_f64/120.0_f64
353  self%mass_0(2) = 26.0_f64/120.0_f64
354  self%mass_0(3) = 1.0_f64/120.0_f64
355  ! Upper diagonal coeficients of linear spline mass matrix (from Eulerian numbers)
356  self%mass_1(1) = 4.0_f64/6.0_f64
357  self%mass_1(2) = 1.0_f64/6.0_f64
358  case (3)
359  ! Upper diagonal coeficients of cubic spline mass matrix (from Eulerian numbers)
360  self%mass_0(1) = 2416.0_f64/5040.0_f64
361  self%mass_0(2) = 1191.0_f64/5040.0_f64
362  self%mass_0(3) = 120.0_f64/5040.0_f64
363  self%mass_0(4) = 1.0_f64/5040.0_f64
364  ! Upper diagonal coeficients of quadratic spline mass matrix (from Eulerian numbers)
365  self%mass_1(1) = 66.0_f64/120.0_f64
366  self%mass_1(2) = 26.0_f64/120.0_f64
367  self%mass_1(3) = 1.0_f64/120.0_f64
368 
369  case default
370  print *, 'sll_t_maxwell_1d_fem init: spline degree ', s_deg_0, ' not implemented'
371  end select
372 
373  sll_allocate(self%eig_mass0(n_dofs), ierr)
374  sll_allocate(self%eig_mass1(n_dofs), ierr)
375  sll_allocate(self%eig_weak_ampere(n_dofs), ierr)
376  sll_allocate(self%eig_weak_poisson(n_dofs), ierr)
377  ! Initialise FFT
378  sll_allocate(self%work(n_dofs), ierr)
379  sll_allocate(self%wsave(n_dofs), ierr)
380  call sll_s_fft_init_r2r_1d(self%plan_fw, self%n_dofs, self%work, self%wsave, sll_p_fft_forward, normalized=.false.)
381  call sll_s_fft_init_r2r_1d(self%plan_bw, self%n_dofs, self%work, self%work, sll_p_fft_backward, normalized=.false.)
382 
383  ! Compute eigenvalues of circulant Ampere update matrix M_0^{-1} D^T M_1
384  ! and circulant Poisson Matrix (D^T M_1 D)^{-1}
385  ! zero mode vanishes due to derivative matrix D^T
386  self%eig_weak_ampere(1) = 0.0_f64
387  self%eig_weak_poisson(1) = 0.0_f64 ! Matrix is not invertible: 0-mode is set to 0
388  self%eig_mass0(1) = 1.0_f64 ! sum of coefficents is one
389  self%eig_mass1(1) = 1.0_f64 ! sum of coefficents is one
390 
391  do k = 1, n_dofs/2 - 1
392  coef0 = self%mass_0(1)
393  coef1 = self%mass_1(1)
394  do j = 1, s_deg_0 - 1
395  cos_mode = cos(2*sll_p_pi*j*k/real(n_dofs, f64))
396  coef0 = coef0 + 2*self%mass_0(j + 1)*cos_mode
397  coef1 = coef1 + 2*self%mass_1(j + 1)*cos_mode
398  end do
399  ! add last term for larger matrix
400  j = s_deg_0
401  coef0 = coef0 + 2*self%mass_0(j + 1)*cos(2*sll_p_pi*j*k/real(n_dofs, f64))
402  ! compute eigenvalues
403  self%eig_mass0(k + 1) = coef0 ! real part
404  self%eig_mass0(n_dofs - k + 1) = 0.0_f64 ! imaginary part
405  self%eig_mass1(k + 1) = coef1 ! real part
406  self%eig_mass1(n_dofs - k + 1) = 0.0_f64 ! imaginary part
407  cos_mode = cos(2*sll_p_pi*k/real(n_dofs, f64))
408  sin_mode = sin(2*sll_p_pi*k/real(n_dofs, f64))
409  self%eig_weak_ampere(k + 1) = (coef1/coef0)*(1 - cos_mode) ! real part
410  self%eig_weak_ampere(n_dofs - k + 1) = -(coef1/coef0)*sin_mode ! imaginary part
411  self%eig_weak_poisson(k + 1) = 1.0_f64/(coef1*((1 - cos_mode)**2 + &
412  sin_mode**2)) ! real part
413  self%eig_weak_poisson(n_dofs - k + 1) = 0.0_f64 ! imaginary part
414  end do
415  ! N/2 mode
416  coef0 = self%mass_0(1)
417  coef1 = self%mass_1(1)
418  do j = 1, s_deg_0 - 1
419  coef0 = coef0 + 2*self%mass_0(j + 1)*cos(sll_p_pi*j)
420  coef1 = coef1 + 2*self%mass_1(j + 1)*cos(sll_p_pi*j)
421  end do
422 
423  ! add last term for larger matrix
424  j = s_deg_0
425  coef0 = coef0 + 2*self%mass_0(j + 1)*cos(sll_p_pi*j)
426 
427  ! compute eigenvalues
428  self%eig_mass0(n_dofs/2 + 1) = coef0
429  self%eig_mass1(n_dofs/2 + 1) = coef1
430  self%eig_weak_ampere(n_dofs/2 + 1) = 2.0_f64*(coef1/coef0)
431  self%eig_weak_poisson(n_dofs/2 + 1) = 1.0_f64/(coef1*4.0_f64)
432 
433  end subroutine init_1d_fem
