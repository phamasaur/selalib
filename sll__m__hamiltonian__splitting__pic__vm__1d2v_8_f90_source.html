139  subroutine operatorhp1_pic_vm_1d2v(self, dt)
140  class(sll_t_hamiltonian_splitting_pic_vm_1d2v), intent(inout) :: self
141  sll_real64, intent(in) :: dt
142 
143  !local variables
144  sll_int32 :: i_part
145  sll_real64 :: x_new(3), vi(3), wi(1), x_old(3)
146  sll_int32 :: n_cells
147  sll_real64 :: qoverm
148 
149  n_cells = self%kernel_smoother_0%n_dofs
150 
151  ! Here we have to accumulate j and integrate over the time interval.
152  ! At each k=1,...,n_grid, we have for s \in [0,dt]:
153  ! j_k(s) = \sum_{i=1,..,N_p} q_i N((x_k+sv_{1,k}-x_i)/h) v_k,
154  ! where h is the grid spacing and N the normalized B-spline
155  ! In order to accumulate the integrated j, we normalize the values of x to the grid spacing, calling them y, we have
156  ! j_k(s) = \sum_{i=1,..,N_p} q_i N(y_k+s/h v_{1,k}-y_i) v_k.
157  ! Now, we want the integral
158  ! \int_{0..dt} j_k(s) d s = \sum_{i=1,..,N_p} q_i v_k \int_{0..dt} N(y_k+s/h v_{1,k}-y_i) ds = \sum_{i=1,..,N_p} q_i v_k \int_{0..dt} N(y_k + w v_{1,k}-y_i) dw
159 
160  self%j_dofs_local = 0.0_f64
161 
162  ! For each particle compute the index of the first DoF on the grid it contributes to and its position (normalized to cell size one). Note: j_dofs(_local) does not hold the values for j itself but for the integrated j.
163  ! Then update particle position: X_new = X_old + dt * V
164  do i_part = 1, self%particle_group%n_particles
165  ! Read out particle position and velocity
166  x_old = self%particle_group%get_x(i_part)
167  vi = self%particle_group%get_v(i_part)
168 
169  ! Then update particle position: X_new = X_old + dt * V
170  x_new = x_old + dt*vi
171 
172  ! Get charge for accumulation of j
173  wi = self%particle_group%get_charge(i_part)
174  qoverm = self%particle_group%species%q_over_m();
175  call self%kernel_smoother_1%add_current_update_v(x_old, x_new, wi(1), qoverm, &
176  self%bfield_dofs, vi, self%j_dofs_local(:, 1))
177  ! Accumulate rho for Poisson diagnostics
178  call self%kernel_smoother_0%add_charge(x_new, wi(1), &
179  self%j_dofs_local(:, 2))
180 
181  x_new(1) = modulo(x_new(1), self%Lx)
182  call self%particle_group%set_x(i_part, x_new)
183  call self%particle_group%set_v(i_part, vi)
184 
185  end do
186 
187  self%j_dofs = 0.0_f64
188  ! MPI to sum up contributions from each processor
189  call sll_o_collective_allreduce(sll_v_world_collective, self%j_dofs_local(:, 1), &
190  n_cells, mpi_sum, self%j_dofs(:, 1))
191 
192  ! Update the electric field.
193  call self%maxwell_solver%compute_E_from_j(self%j_dofs(:, 1), 1, self%efield_dofs(:, 1))
194 
195  end subroutine operatorhp1_pic_vm_1d2v
